    JedisSentinelPool pool = new JedisSentinelPool("mymaster", IPS);
    Jedis jedis=null;
    try{
        jedis = pool.getResource();
        jedis.set("foo", "bar");
        System.out.println(jedis.get("foo"));
    } catch(Exception e){
